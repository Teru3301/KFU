Дан кода на Go. Что будет выведено на экран в результате его выполнения:

package main
import "fmt"

func change(abc []int) {
    for i := range abc {
        abc[i] = 4
    }
    fmt.Println(abc)
}

func main() {
    abc := []int{1, 2, 3}
    change(abc)
    fmt.Println(abc)
}

Ответы:
- [ ]
    [1 2 3][1 2 3]
- [ ]
    [4 4 4][1 2 3]
- [x]
    [4 4 4][4 4 4]
- [ ]
    [1 2 4][1 2 3]
